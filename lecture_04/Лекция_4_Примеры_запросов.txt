select
	to_char(current_timestamp, 'HH:MI:SS')
  , to_char(125, '9999')
  , to_char(-125.8, '999D99S')
  , to_date('05 Dec 2000', 'DD Mon YYYY')
  , to_number('12,454.8-', '99G999D9S')
  , to_timestamp('05 Dec 2000', 'DD Mon YYYY');

    *
  , nullif(job_title, 'President')
from jobs;

select
	region_id
  , region_name
  , case 
        when region_name = 'Asia' THEN 'Азия'
	    else 'Не Азия'
	end
  , case region_name
        when 'Asia' then 'Азия'
		else 'Не Азия'
	end
from regions;

select
    first_name
  , last_name
  , salary
  , case
        when salary between 0.00 and 4999.99
            then 'Low salary'
	    when salary between 5000.00 and 9999.99
	        then 'Normal salary'
		when salary between 10000.00 and 14999.99
	        then 'High salary'
		else 'TOP'
	end
from employees
order by employee_id
limit 10;

select
	emp.first_name
  , emp.last_name
  , mgr.first_name
  , mgr.last_name
from employees as emp
left join employees as mgr
    on emp.manager_id = mgr.employee_id;
	

select
    emp.employee_id
  , emp.first_name
  , emp.last_name
  , emp.salary
  , emp.job_id
  , jobs.job_title
from employees as emp
left join jobs
on emp.job_id = jobs.job_id and
   emp.salary between jobs.min_salary and jobs.max_salary;


select
    emp.first_name as emp_first_name
  , emp.last_name as emp_last_name
  , mgr.first_name as mgr_first_name
  , mgr.last_name as mgr_last_name
from employees as emp
left join employees as mgr
    on emp.manager_id = mgr.employee_id;
	

select
    emp.first_name as emp_first_name
  , emp.last_name as emp_last_name
  , mgr.first_name as mgr_first_name
  , mgr.last_name as mgr_last_name
from employees as emp
right join employees as mgr
    on emp.manager_id = mgr.employee_id;
	
	
select
    emp.first_name as emp_first_name
  , emp.last_name as emp_last_name
  , mgr.first_name as mgr_first_name
  , mgr.last_name as mgr_last_name
from employees as emp
full join employees as mgr
    on emp.manager_id = mgr.employee_id;

select employee_id, last_name, job_id, salary
from   employees
where  salary < any
                    (select salary
                     from   employees
                     where  job_id = 6)
AND    job_id <> 6;


select employee_id, last_name, job_id, salary
from   employees
where  salary < all
                    (select salary
                     from   employees
                     where  job_id = 6)
and job_id <> 6;
