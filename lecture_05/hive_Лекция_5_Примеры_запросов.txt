select
	region_id
  , region_name
  , case 
        when region_name = 'Asia' THEN 'Азия'
	    else 'Не Азия'
	end
  , case region_name
        when 'Asia' then 'Азия'
		else 'Не Азия'
	end
from hr.regions;

select
    first_name
  , last_name
  , salary
  , case
        when salary between 0.00 and 4999.99
            then 'Low salary'
	    when salary between 5000.00 and 9999.99
	        then 'Normal salary'
		when salary between 10000.00 and 14999.99
	        then 'High salary'
		else 'TOP'
	end
from hr.employees
order by employee_id
limit 10;

-- Неявная проверка на вхождение зарплаты сотрудника в настроенный диапазон 
select
    emp.employee_id
  , emp.first_name
  , emp.last_name
  , emp.salary
  , emp.job_id
  , jobs.job_title
from hr.employees emp
left join hr.jobs
on emp.job_id = jobs.job_id and
   emp.salary  between jobs.min_salary and jobs.max_salary;



-- Фильтрация по зарплате в конце запроса, но с выборкой детальной информации из итбалицы jobs
select
    emp.employee_id
  , emp.first_name
  , emp.last_name
  , emp.salary
  , emp.job_id
  , jobs.job_title
from hr.employees emp
left join hr.jobs
on emp.job_id = jobs.job_id
where emp.salary > 10000;

-- Вывести сотрудника, который является менеджером и с зарплатой более N
select  employee_id,
		salary,
		last_name 
from hr.employees t1
where exists
			(select employee_id 
			  from hr.employees t2
			 where t2.manager_id = t1.employee_id
			 and t2.salary > 10000);
