
CREATE TABLE regions (
    region_id INT,
    region_name STRING
)
STORED AS ORC;

CREATE TABLE countries (
    country_id STRING,
    country_name STRING,
    region_id INT
)
STORED AS ORC;

CREATE TABLE locations (
    location_id INT,
    street_address STRING,
    postal_code STRING,
    city STRING,
    state_province STRING,
    country_id STRING
)
STORED AS ORC;

CREATE TABLE departments (
    department_id INT,
    department_name STRING,
    location_id INT
)
STORED AS ORC;

CREATE TABLE jobs (
    job_id INT,
    job_title STRING,
    min_salary DECIMAL(8,2),
    max_salary DECIMAL(8,2)
)
STORED AS ORC;

CREATE TABLE employees (
    employee_id INT,
    first_name STRING,
    last_name STRING,
    email STRING,
    phone_number STRING,
    hire_date DATE,
    job_id INT,
    salary DECIMAL(8,2),
    manager_id INT,
    department_id INT
)
STORED AS ORC;

CREATE TABLE dependents (
    dependent_id INT,
    first_name STRING,
    last_name STRING,
    relationship STRING,
    employee_id INT
)
STORED AS ORC;
